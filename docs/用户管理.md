1. 权限装饰器（Bot端）
from functools import wraps
from database.models import UserRole

def require_role(*allowed_roles: UserRole):
    """要求特定角色才能访问"""
    def decorator(func):
        @wraps(func)
        async def wrapper(message: Message, user: User, *args, **kwargs):
            if user.role not in allowed_roles:
                await message.answer(
                    "❌ 权限不足\n"
                    f"需要角色: {', '.join([r.value for r in allowed_roles])}"
                )
                return
            return await func(message, user, *args, **kwargs)
        return wrapper
    return decorator

# 使用 CTE 优化复杂统计查询
from sqlalchemy import func, case

async def get_user_statistics(db: AsyncSession):
    # 角色分布
    role_dist = await db.execute(
        select(
            User.role,
            func.count(User.id).label('count')
        ).group_by(User.role)
    )
    
    # 活跃用户（最近7天）
    seven_days_ago = datetime.now() - timedelta(days=7)
    active_users = await db.scalar(
        select(func.count(User.id))
        .where(User.last_active_at >= seven_days_ago)
    )
    
    return {
        "role_distribution": dict(role_dist.all()),
        "active_users_7d": active_users
    }
