# 权限管理文档

## 概述

权限管理系统负责控制用户对 Bot 功能和合集的访问权限，包括用户角色管理、管理员权限管理、访问控制、封禁机制等。

## 用户角色体系

### 角色定义

| 角色 | 英文标识 | 权限描述 |
|------|---------|---------|
| 普通用户 | `user` | 访问公开合集、搜索功能 |
| VIP 用户 | `vip` | 访问公开合集 + VIP 合集 |
| 管理员 | `admin` | 所有用户权限 + 管理功能（上传、编辑、删除合集） |
| 超级管理员 | `super_admin` | 所有权限 + 用户管理 + 系统设置 |

### 权限矩阵

| 功能 | user | vip | admin | super_admin |
|------|------|-----|-------|-------------|
| 访问公开合集 | ✅ | ✅ | ✅ | ✅ |
| 访问 VIP 合集 | ❌ | ✅ | ✅ | ✅ |
| 搜索合集 | ✅ | ✅ | ✅ | ✅ |
| 上传合集 | ❌ | ❌ | ✅ | ✅ |
| 编辑合集 | ❌ | ❌ | ✅ | ✅ |
| 删除合集 | ❌ | ❌ | ✅ | ✅ |
| 创建搬运任务 | ❌ | ❌ | ✅ | ✅ |
| 管理用户权限 | ❌ | ❌ | ✅ | ✅ |
| 管理管理员 | ❌ | ❌ | ❌ | ✅ |
| 系统设置 | ❌ | ❌ | ✅ | ✅ |
| 访问 Web 后台 | ❌ | ✅ | ✅ | ✅ |

## 权限检查机制

### 中间件实现

使用 aiogram 中间件在每个请求前检查权限：

```python
from aiogram import BaseMiddleware
from aiogram.types import Message, CallbackQuery

class AuthMiddleware(BaseMiddleware):
    """权限检查中间件"""

    async def __call__(self, handler, event, data):
        # 获取用户信息
        if isinstance(event, Message):
            user_id = event.from_user.id
        elif isinstance(event, CallbackQuery):
            user_id = event.from_user.id
        else:
            return await handler(event, data)

        # 查询用户
        user = await get_user_by_telegram_id(user_id)

        if not user:
            # 自动注册新用户
            user = await register_user(
                telegram_id=user_id,
                username=event.from_user.username,
                first_name=event.from_user.first_name,
                last_name=event.from_user.last_name
            )

        # 检查封禁状态
        if user.is_banned:
            if isinstance(event, Message):
                await event.answer("🚫 您已被封禁，无法使用 Bot")
            return

        # 将用户信息注入到 handler
        data["user"] = user

        return await handler(event, data)
```

### 装饰器实现

为特定功能添加权限检查装饰器：

```python
from functools import wraps

def require_role(*allowed_roles: str):
    """要求特定角色才能访问"""
    def decorator(func):
        @wraps(func)
        async def wrapper(message: Message, user: User, *args, **kwargs):
            if user.role not in allowed_roles:
                await message.answer(
                    "❌ 权限不足\n"
                    f"需要角色: {', '.join(allowed_roles)}"
                )
                return

            return await func(message, user, *args, **kwargs)
        return wrapper
    return decorator
```

### 合集访问权限检查

```python
async def check_collection_access(
    user: User,
    collection: Collection
) -> tuple[bool, str]:
    """
    检查用户是否有权限访问合集

    返回: (是否有权限, 错误消息)
    """
    # 公开合集所有人可访问
    if collection.access_level == "public":
        return True, ""

    # VIP 合集需要 VIP 权限
    if collection.access_level == "vip":
        if user.role in ["vip", "admin", "super_admin"]:
            return True, ""
        else:
            return False, (
                "❌ 这是 VIP 专属合集\n"
                "💎 升级为 VIP 用户即可访问\n\n"
                "联系管理员获取 VIP 权限"
            )

    return False, "❌ 未知的访问权限类型"
```

### 设置用户角色

```python
@router.message(Command("setrole"))
@require_role("super_admin")
async def set_user_role(message: Message, user: User):
    """设置用户角色"""
    args = message.text.split()
    if len(args) < 3:
        await message.answer(
            "用法: /setrole {telegram_id} {role}\n"
            "角色: user, vip, admin, super_admin"
        )
        return

    target_id = int(args[1])
    new_role = args[2]

    # 验证角色
    valid_roles = ["user", "vip", "admin", "super_admin"]
    if new_role not in valid_roles:
        await message.answer(f"❌ 无效的角色: {new_role}")
        return

    # 更新角色
    result = await db.execute(
        update(User)
        .where(User.telegram_id == target_id)
        .values(role=new_role)
    )
    await db.commit()

    if result.rowcount == 0:
        await message.answer("❌ 用户不存在")
        return

    # 记录日志
    await log_admin_action(
        user_id=user.id,
        action="set_user_role",
        details={
            "target_id": target_id,
            "new_role": new_role
        }
    )

    await message.answer(f"✅ 已将用户 {target_id} 的角色设置为 {new_role}")
```

```python
async def ban_user(message: Message, user: User):
    """封禁用户"""
    args = message.text.split()
    if len(args) < 2:
        await message.answer("用法: /ban {telegram_id}")
        return

    # 不能封禁自己

    await db.execute(
        update(User)
        .where(User.telegram_id == target_id)
        .values(is_banned=True)
    )
    await db.commit()

    # 记录日志
    await log_admin_action(
        user_id=user.id,
        action="ban_user",
        details={"target_id": target_id}
    )

    await message.answer(f"✅ 已封禁用户 {target_id}")
```

### 批量设置 VIP

**命令**: `/batch_vip` (仅超级管理员)

**流程**:
```
/batch_vip
  ↓
Bot: "请发送用户 ID 列表（每行一个）"
  ↓
管理员:
  123456789
  987654321
  111222333
  ↓
Bot: "确认将 3 个用户设置为 VIP？"
  [确认] [取消]
  ↓
管理员: 点击 [确认]
  ↓
Bot: "✅ 已设置 3 个用户为 VIP"
```

```python
class BatchVIPStates(StatesGroup):
    waiting_for_ids = State()

@router.message(Command("batch_vip"))
@require_role("super_admin")
async def batch_vip_start(message: Message, state: FSMContext):
    """批量设置 VIP"""
    await message.answer(
        "📝 请发送用户 ID 列表（每行一个）\n"
        "发送 /cancel 取消"
    )
    await state.set_state(BatchVIPStates.waiting_for_ids)

@router.message(BatchVIPStates.waiting_for_ids)
async def batch_vip_process(message: Message, state: FSMContext):
    """处理批量 VIP"""
    user_ids = []
    for line in message.text.strip().split('\n'):
        try:
            user_ids.append(int(line.strip()))
        except ValueError:
            continue

    if not user_ids:
        await message.answer("❌ 没有有效的用户 ID")
        return

    # 确认
    keyboard = InlineKeyboardMarkup(inline_keyboard=[
        [
            InlineKeyboardButton(text="✅ 确认", callback_data=f"batch_vip_confirm"),
            InlineKeyboardButton(text="❌ 取消", callback_data="batch_vip_cancel")
        ]
    ])

    await state.update_data(user_ids=user_ids)
    await message.answer(
        f"确认将 {len(user_ids)} 个用户设置为 VIP？",
        reply_markup=keyboard
    )

@router.callback_query(F.data == "batch_vip_confirm")
async def batch_vip_confirm(callback: CallbackQuery, state: FSMContext):
    """确认批量设置 VIP"""
    data = await state.get_data()
    user_ids = data.get("user_ids", [])

    # 批量更新
    await db.execute(
        update(User)
        .where(User.telegram_id.in_(user_ids))
        .values(role="vip")
    )
    await db.commit()

    await state.clear()
    await callback.message.answer(f"✅ 已设置 {len(user_ids)} 个用户为 VIP")
```

## Web 后台权限

### Telegram Login 认证

使用 Telegram Login Widget 进行 Web 后台登录：

```python
from fastapi import APIRouter, HTTPException, Depends
from fastapi.security import HTTPBearer
import hashlib
import hmac

router = APIRouter()
security = HTTPBearer()

def verify_telegram_auth(auth_data: dict, bot_token: str) -> bool:
    """验证 Telegram Login 数据"""
    check_hash = auth_data.pop("hash")

    # 构建数据字符串
    data_check_string = "\n".join(
        f"{k}={v}" for k, v in sorted(auth_data.items())
    )

    # 计算哈希
    secret_key = hashlib.sha256(bot_token.encode()).digest()
    calculated_hash = hmac.new(
        secret_key,
        data_check_string.encode(),
        hashlib.sha256
    ).hexdigest()

    return calculated_hash == check_hash

@router.post("/auth/telegram")
async def telegram_login(auth_data: dict):
    """Telegram 登录"""
    # 验证数据
    if not verify_telegram_auth(auth_data, BOT_TOKEN):
        raise HTTPException(status_code=401, detail="认证失败")

    # 获取用户
    user = await get_user_by_telegram_id(auth_data["id"])
    if not user:
        raise HTTPException(status_code=403, detail="用户不存在")

    # 检查权限（仅管理员可登录）
    if user.role not in ["admin", "super_admin"]:
        raise HTTPException(status_code=403, detail="权限不足")

    # 生成 JWT token
    token = create_access_token(
        data={"user_id": user.id, "role": user.role}
    )

    return {
        "access_token": token,
        "token_type": "bearer",
        "user": {
            "id": user.telegram_id,
            "username": user.username,
            "first_name": user.first_name,
            "role": user.role
        }
    }
```

## 安全建议

1. **最小权限原则**: 仅授予必要的权限
2. **定期审计**: 定期检查管理员列表和操作日志
3. **双因素认证**: 考虑为超级管理员启用两步验证
4. **IP 白名单**: 限制 Web 后台访问 IP（可选）
5. **会话管理**: JWT token 设置合理的过期时间
6. **操作确认**: 敏感操作（删除、封禁）需要二次确认
